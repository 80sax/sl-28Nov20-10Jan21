## Pipeline Structure

    pipeline {
        agent
        tools
        stages {
            stage1 {
                step1 {}
                Step2 {}
            }
            stage2 {
                step1 {}
                Step2 {}
            }
            stage3 {
                step1 {}
                Step2 {}
            }
        }
    }

## Pipeline Example

    pipeline {
        agent any
        tools {
            maven 'maven3'
        }
        stages {
            stage('Hello') {
                steps {
                    echo 'Hello World'
                }
            }
            stage('Checkout') {
                steps {
                    git 'https://github.com/sk12k/addressbook.git'
                }
            }
                    stage ('build') {
            steps {
                sh '/var/lib/jenkins/tools/hudson.tasks.Maven_MavenInstallation/maven3/bin/mvn -f pom.xml package'
            }
        }
    }
